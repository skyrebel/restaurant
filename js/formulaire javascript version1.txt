var formValid = document.getElementById('submit-button');

var name = document.getElementById('fullname');
var missfullname = document.getElementById('missfullname');
var nameValid = /^[a-zA-Zéèîï][a-zéèêàçîï]+([-'\s][a-zA-Zéèîï][a-zéèêàçîï]+)?$/;

var mail = document.getElementById('mailadress');
var missemail = document.getElementById('missemail');
var mailValid = /^[a-zA-Zéèîï][a-zéèêàçîï]+([-'\s][a-zA-Zéèîï][a-zéèêàçîï]+)?$/;

var subject = document.getElementById('msubject');
var misssubject = document.getElementById('misssubject');
var subjectValid = /^[a-zA-Zéèîï][a-zéèêàçîï]+([-'\s][a-zA-Zéèîï][a-zéèêàçîï]+)?$/;

var message = document.getElementById('message');
var missmessage = document.getElementById('missmessage');
var messageValid = /^[a-zA-Zéèîï][a-zéèêàçîï]+([-'\s][a-zA-Zéèîï][a-zéèêàçîï]+)?$/;



formValid.addEventListener('click', validation);

function validation(event){

    //Si les champs sont vides
    if (name.validity.valueMissing){
        event.preventDefault();
        missfullname.textContent = 'Please enter your full name.';
        missfullname.style.color = 'white';
    }

    if (mail.validity.valueMissing){
        event.preventDefault();
        missfullname.textContent = 'Please enter your email.';
        missfullname.style.color = 'white';
    }

    if (subject.validity.valueMissing){
        event.preventDefault();
        missfullname.textContent = 'Please enter the subject.';
        missfullname.style.color = 'white';
    }

    if (message.validity.valueMissing){
        event.preventDefault();
        missfullname.textContent = 'Please enter your message.';
        missfullname.style.color = 'white';
        // Si le format des données est incorrect
    }
    
    else if (nameValid.test(name.value) == false){
        event.preventDefault();
        missfullname.textContent = 'Please enter a valid name.';
        missfullname.style.color = 'white';
    }

    else if (mailValid.test(mail.value) == false){
        event.preventDefault();
        missfullname.textContent = 'Please enter a valid name.';
        missfullname.style.color = 'white';
    }

    else if (subjectValid.test(subject.value) == false){
        event.preventDefault();
        missfullname.textContent = 'Please enter a valid name.';
        missfullname.style.color = 'white';
    }

    else if (messageValid.test(message.value) == false){
        event.preventDefault();
        missfullname.textContent = 'Please enter a valid name.';
        missfullname.style.color = 'white';
    }

}
